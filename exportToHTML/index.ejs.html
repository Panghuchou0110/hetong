443      function isValidChineseId(value) {
444        const id = String(value || "").toUpperCase();
445        if (!/^\d{17}[\dX]$/.test(id)) return false;
446        const weights = [7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2];
447        const checkMap = ["1", "0", "X", "9", "8", "7", "6", "5", "4", "3", "2"];
448        let sum = 0;
449        for (let i = 0; i < 17; i += 1) {
450          sum += Number(id[i]) * weights[i];
451        }
452        return checkMap[sum % 11] === id[17];
453      }
